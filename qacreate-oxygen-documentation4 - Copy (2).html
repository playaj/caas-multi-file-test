<xsl:template name="get-topic-title-id">
  <xsl:variable name="topicTitle01">
    <xsl:value-of select="td:replacetmreg(preceding::*[contains(@class, ' topic/title ')][1])"/>
  </xsl:variable>
  <xsl:variable name="topicTitle">
    <xsl:value-of select="td:replacetmreg($topicTitle01)"/>
  </xsl:variable>
  <data type="topicreport" name="ixiasoftID">
    <xsl:choose>
      <xsl:when test="contains(ancestor::*[contains(@class, ' topic/topic ')][1]/@id, 'unique')">
        <xsl:value-of select="' '"/>
      </xsl:when>
      <xsl:when test="contains(ancestor::*[contains(@class, ' topic/topic ')][1]/@id, ' ')">
        <xsl:value-of select="substring-after(ancestor-or-self::*[contains(@class, ' topic/topic ')][1]/@id, ' ')"/>
      </xsl:when>
      <xsl:otherwise>
        <xsl:value-of select="ancestor::*[contains(@class, ' topic/topic ')][1]/@id"/>
      </xsl:otherwise>
    </xsl:choose>
  </data>
  <data type="topicreport" name="topicTitle">
    <xsl:value-of select="$topicTitle"/>
  </data>
</xsl:template>
